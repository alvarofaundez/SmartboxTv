 24     /**
 25      * Helper function to properly handler http requests
 26      *
 27      * @param req {XMLHTTPRequest}   the request object
 28      * @param callback {function}       the function to call with the response
 29      */
 30     _requestHelper = function _requestHelper(req, callback){
 31         if (req.readyState === 4) {
 32             if (req.status >= 200 && req.status < 400) {
 33                 // Return the responseText and the responseXML
 34                 var headers;
 35                 if (typeof req.getAllResponseHeaders !== 'undefined'){
 36                 	headers = req.getAllResponseHeaders();
 37                 }
 38                 callback(req.responseText, req.responseXML || req.responseValue, headers);
 39 
 40 	    // Local files
 41             }  else if ( req.status === 0 ) {
 42 
 43 		callback(req.responseText, req.responseXML, {});	    
 44 	    
 45 	    } else {
 46                 var error = new Error('HTTP Error code: ' + req.status);
 47                 error.type = 'error';
 48                 error.code = req.status;
 49                 error.message = 'HTTP Error code: ' + req.status;
 50                 error.body = req.responseText;
 51                 callback(error);
 52             }
 53         }
 54     };

134     /**
135      * A flexible simple cross browser event attacher.
136 	 * Registers a function callback which will be called when the given event occurs.
137 	 * It allows to register event handlers for typical clicks, mouseovers... and also for level changes, setting updates... (by passing Pincho as the first argument).
138 	 * 
139      * @name hook
140 	 * @function
141 	 * @memberOf Pincho
142      * @param {node} obj                the DOM object to which attach the event or the SDK for custom events
143      * @param {string} eventName        the event name (without the 'on' prefix)
144      * @param {function} eventHandler   the function to launch on the event
145      * @param {object} context          the context in which the function will be called
146      * @returns {function}  To be used for unhook()
147 	 * @example Pincho.hook(Pincho, "level", changeLevel);
148 	 * @example Pincho.hook(Pincho, "key", processKey);
149      */
150     self.hook = function( obj, eventName, eventHandler, context ){
151         // Adjust the arguments if the 'obj' argument was not passed
152         if ( typeof obj === 'string' ) {
153             context = eventHandler;
154             eventHandler = eventName;
155             eventName = obj;
156             obj = self;
157         }
158 
159         // If no context supplied then use "window"
160         context = context || window;
161 
162         // Create a function override to apply the given context
163         var func = function(e){
164             e = e || window.event;
165             return eventHandler.call(context, e);
166         };
167 
168         // If the event is register to the SDK (custom events)
169         if ( obj === self ) {
170             if ( typeof _eventHandlers[eventName] === 'undefined' ) {
171                 _eventHandlers[eventName] = [ func ];
172             } else {
173                 _eventHandlers[eventName].push( func );
174             }
175 
176         } else if ( obj.attachEvent ) {
177 
178             obj.attachEvent('on' + eventName, func );
179 
180         } else if ( obj.addEventListener ) {
181 
182             obj.addEventListener(eventName, func, false);
183 
184         }
185 	    return func;
186     };

189     /**
190      * Remove an event handler (unhook)
191 	 * 
192      * @name unhook
193 	 * @function
194 	 * @memberOf Pincho
195      * @param {node} obj                the DOM object to which attach the event or the ICUE singleton for custom events
196      * @param {string} eventName       the event name (without the 'on' prefix)
197      * @param {function} eventHandler  the function to launch on the event
198 	 * @example Pincho.unhook(Pincho, "key", processKey);
199      */
200     self.unhook = function( obj, eventName, eventHandler ){
201         // Adjust the arguments if 'obj' is omitted
202         if ( typeof obj === 'string' ) {
203             eventHandler = eventName;
204             eventName = obj;
205             obj = self;
206         }
207 
208         // unhook custom events
209         if ( obj === self ) {
210 
211             if ( typeof _eventHandlers[eventName] === 'undefined' ) {
212                 return false;
213             }
214 
215             var k = _eventHandlers[eventName].length;
216             while (k--) {
217                 if ( _eventHandlers[eventName][k] === eventHandler ) {
218                     _eventHandlers[eventName].splice(k,1);
219                     return true;
220                 }
221             }
222 
223         } else if ( obj.detachEvent ) {
224 
225             return obj.detachEvent('on' + eventName, eventHandler );
226 
227         } else if ( obj.removeEventListener ) {
228 
229             return obj.removeEventListener(eventName, eventHandler, false);
230 
231         }
232 
233         return false;
234     };

237     /**
238      * Trigger custom events.
239 	 * For the time being, this function is only used by the services to notify the container that the preview has been loaded. 
240 	 * This function has to be called when all the elements in the preview have been loaded.
241 	 * <p>In the following example the parameter typePreview is optional. The possible values are 1,2 or 3: 
242 	 * <ul>
243 	 * <li>1 for static preview (is the default value), </li>
244 	 * <li>2 for dinamic general preview (only loaded when the container is initialized), </li>
245 	 * <li>3 for dinamic customized preview (loaded when the container is initialized and when the function Pincho.reloadPreview() has been called).</li>
246 	 * </ul>
247 	 * 
248      * @name trigger
249 	 * @function
250 	 * @memberOf Pincho
251      * @param {string} eventName    Name of the event
252      * @param {mixed} arg          Argument to pass into the handler
253      * @returns {bool}  if False then the propagation has been stopped
254 	 * @example Pincho.trigger(‘preview’,typePreview);
255      */
256     self.trigger = function(eventName, arg){
257         var ret;
258 
259         // If it's a 'level' event then store the new level
260         if( eventName == "level" ) {
261             _level = arg;
262         }
263 
264         if ( typeof _eventHandlers[eventName] !== 'undefined' ) {
265             var k = _eventHandlers[eventName].length;
266             // For each registered event launch the handler until one returns
267             // false or all of them have been executed
268             while (k-- && ret !== false) {
269                 ret = _eventHandlers[eventName][k](arg);
270             }
271 
272         }
273 		
274         return ret;
275     };

320     /**
321      * Requests a json document and automatically decodes it.
322 	 * 
323      * @name json.request
324 	 * @function
325 	 * @memberOf Pincho
326 	 * @requires json2.js
327      * @param {string} url              The remote resource
328      * @param {mixed} params           A list of key-value pairs or the request body
329      * @param {functions} onResponse   The function to call when done
330      * @param {string} verb            (optional) The http verb (method) to use (GET by default)
331      * @param {object} headers         (optional) A hash object with the request headers
332      * @param {string} user            (optional) The user name for authentication
333      * @param {string} pass            (optional) The password for authentication
334      */
335     self.json.request = function(url, params, onResponse, verb, headers, user, pass){
336         // Wrap the supplied callback to automatically decode json
337         var jsonResponse = function(data){
338             if ( typeof data === 'string' ) {
339                 data = self.json.decode(data);

343         }

349     self.request = function(url, params, onResponse, verb, headers, user, pass, async){
350         var k;
351         
352         var asyncReq = typeof async != 'undefined' ? async : true;
353         
354         // Use GET by default
355         verb = verb ? verb.toUpperCase() : 'GET';
356         
357         if ( typeof params === 'object' ) {
358             var encParams = [];
359             for (k in params){
360                 if (params.hasOwnProperty(k)) {
361                     encParams.push( encodeURIComponent(k) + '=' + encodeURIComponent(params[k]) );
362                 }
363             }
364             params = encParams.join('&');
365         }
366         
367         Pincho.log('MAKING HTTP PETITION', url, params);
368         
369         if ( verb === 'GET' && params.length) {
370             url = url.replace(/#.*$/,'');
371             url += url.indexOf('?') >= 0 ? '&' : '?';
372             url += params;
373         }
374         
375         var req = new XMLHttpRequest();
376         
377         // Attach our helper to the onreadystatechange
378         if (asyncReq == true){
379             req.onreadystatechange = function(){
380                 _requestHelper( req, onResponse );
381             };
382         }
383         
384         // Configure the request
385         req.open(verb, url, asyncReq, user, pass);
386         
387         // Process the headers
388         if (headers) {
389             for (k in headers){
390                 if (headers.hasOwnProperty(k)){
391                     Pincho.log(k, headers[k]);
392                     req.setRequestHeader(k, headers[k]);
393                 }
394             }
395         }
396         
397         // Perform the request
398         if ( typeof params !== 'string' ) {
399             params = '';
400         }
401         req.send(params);
402         
403         // If the petition is asynchronous, call manually the callback function
404         if (asyncReq == false){
405             _requestHelper(req, onResponse);
406         }
407     };

409 	/**
410      * Checks if the url is well-formed and returns an json with the protocol used, domain, port, path, arguments and a hash.
411      * 
412 	 * @name parseUrl
413 	 * @function
414 	 * @memberOf Pincho
415 	 * @param {String} url The url to be parsed
416 	 * @return {mixed} output A json object which contains the protocol, domain, port, path, arguments and a hash. 
417 	 */ 
418     self.parseUrl = function(url){
419 
420         var re = new RegExp('(https?\\:)\\/\\/' +
421                             '([^:\\/\\?\\#]+)' +
422                             ':?(\\d*)' +
423                             '([^\\?\\#]*)' +
424                             '\\??([^\\#]*)' +
425                             '\\#?(.*)'
426                             , 'i');
427         var m = re.exec(url);
428         if (m == null) {
429             
430             // URL sin dominio
431             Pincho.log('Invalid URL');
432             return null;
433         } else {
434 
435             var output = {
436                 'protocol' : m[1],
437                 'domain' : m[2],
438                 'port' : m[3] ? m[3] : (m[1] == 'http' ? 80 : 443 ),
439                 'path' : m[4],
440                 'args' : m[5],
441                 'hash' : m[6]
442             };
443 
444             //Compute args
445             var args = {};
446             var foo = output.args.split('&');
447             for (var i in foo){
448                 if (foo.hasOwnProperty(i)){
449                     var bar = foo[i].split('=');
450                     if (bar.length == 2){
451                         args[decodeURIComponent(bar[0])] = decodeURIComponent(bar[1]);
452                     }
453                 }
454             }
455             output.args = args;
456         }
457         return output;
458     };

574     /**
575      *
576      * Synchronous external javascript loader code by 1mdm ( http://1m.kiev.ua/ )
577      * @link http://forum.mootools.net/viewtopic.php?id=928
578      *
579 	 * @name require
580 	 * @function
581 	 * @memberOf Pincho
582      * @param {string} file The name of the required file.
583      */
584     self.require = function(file) {
585         if (!self.require.loaded[file]) {
586             if (self.require.request) {
587                 var req = self.require.request;
588                 // synchronous request!
589                 req.open('GET', file, false);
590                 req.send(null);
591                 if (req.status == 200) {
592                     if (window.execScript) {
593                         window.execScript(req.responseText, 'javascript');
594                     } else {
595                         window.eval(req.responseText);
596                     }
597                     self.require.loaded[file] = true;
598                 }
599             } else {
600 				self.include(file);
601 			}
602         }
603     };
