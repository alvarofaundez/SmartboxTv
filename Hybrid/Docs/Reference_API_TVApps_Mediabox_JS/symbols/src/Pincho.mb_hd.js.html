636     // Extend request method to go via proxy if the request it's to external resources
637     var tmpFunction = Pincho.request;

639     /**
640      * Peforms request on a remote resource.
641      *
642      * 
643 	 * @name request
644 	 * @function
645 	 * @memberOf Pincho
646      * @param   {string}      url         The remote resource
647      * @param   {mixed}       params      A list of key-value pairs or the request body
648      * @param   {function}    onResponse  The function to call when done
649      * @param   {string}      verb        (optional) The http verb (method) to use (GET by default)
650      * @param   {object}      headers     (optional) A list of key-value pairs
651      * @param   {string}     user       (optional) A username for HTTP Auth
652      * @param   {string}     pass       (optional) A password for HTTP Auth
653      */
654     Pincho.request = function tv_PinchoRequest(url, params, onResponse, verb, headers, user, pass){
655         var parsedUrl = Pincho.parseUrl(url);
656 
657         //If it's and absolute PATH, preppend the BASE_URL
658         if (url.indexOf('/') == 0){
659             url = Pincho.BASE_URL + url;
660         } 
661  
662         // If proxy is needed
663         if ( parsedUrl && ( document.location.protocol != parsedUrl.protocol ||
664             document.location.hostname != parsedUrl.host )) {
665             // Merge GET params on URL to pass it to the proxy
666             if (params && (!verb || verb.toLowerCase() === 'get' || verb.toLowerCase() === 'head')) {
667                 var getParams = [];
668                 for (var i in params) {
669                     if (params.hasOwnProperty(i)){
670                         getParams[getParams.length] = encodeURIComponent(i) + '=' + encodeURIComponent(params[i]);
671                     }
672                 }
673                 if (url.indexOf('?') != -1) {
674                     url += '&' + getParams.join('&');
675                 } else {
676                     url += '?' + getParams.join('&');
677                 }
678                 params = '';
679             // var finalUrl = Pincho.PROXY_URL;
680             }
681             // Adapt the URL to use the proxy
682             url = Pincho.PROXY_URL + '?url=' + encodeURIComponent(url);
683         }
684         tmpFunction.call(Pincho, url, params, onResponse, verb, headers, user, pass);
685     };
